// server.jsconst http = require('http');const hostname = '0.0.0.0';const port = 3000;const server = http.createServer((req, res) => {  res.statusCode = 200;  res.setHeader('Content-Type', 'text/plain');  res.end('Hello, World!\n');});server.listen(port, hostname, () => {  console.log(`Server running at http://${hostname}:${port}/`);FROM node:latestWORKDIR /usr/src/appCOPY . .EXPOSE 3000CMD ["node", "server.js"]docker build -t my-node-app .docker run -p 3000:3000 my-node-app});
